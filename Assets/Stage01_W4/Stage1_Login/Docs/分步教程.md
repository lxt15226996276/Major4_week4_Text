### 3.1 LoginController

| # | 操作 |
|:-:|------|
| 1 | `Scripts/LoginController.cs` 新建脚本 |
| 2 | 挂到 **LoginCanvas** 根 |
| 3 | Inspector 拖齐 InputField · Button 引用 |
| 4 | Play：注册 → 登录成功 → 跳 `SceneNames.Loading`；失败 → **仅 Log** |

#### 本段思路

1. `Dictionary<string,string>` 存账号密码。  
2. 注册：Trim 后非空才写入；重复账号 Log 提示。  
3. 登录：比对 Dictionary；成功 `LoadScene(SceneNames.Loading)`；失败 `Debug.Log("登录失败")`。  
4. `OnDestroy` 里 `RemoveListener` 防泄漏。

#### 带注释 P-L1 范本

```csharp
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

/// <summary>
/// W4 Stage1 登录注册：Dictionary 存账号，成功跳转 Loading。
/// </summary>
public class LoginController : MonoBehaviour
{
    [SerializeField] private InputField _inputAccount;
    [SerializeField] private InputField _inputPassword;
    [SerializeField] private Button _btnLogin;
    [SerializeField] private Button _btnRegister;

    // 内存账号表：用户名 -> 密码（考试项目够用；商业用服务端）
    private readonly Dictionary<string, string> _accounts = new Dictionary<string, string>();

    private void Start()
    {
        _btnLogin.onClick.AddListener(OnLoginClicked);
        if (_btnRegister != null)
            _btnRegister.onClick.AddListener(OnRegisterClicked);
    }

    private void OnDestroy()
    {
        _btnLogin.onClick.RemoveListener(OnLoginClicked);
        if (_btnRegister != null)
            _btnRegister.onClick.RemoveListener(OnRegisterClicked);
    }

    private void OnRegisterClicked()
    {
        string user = _inputAccount.text.Trim();
        string pwd = _inputPassword.text.Trim();
        if (string.IsNullOrEmpty(user) || string.IsNullOrEmpty(pwd))
        {
            Debug.Log("注册失败：用户名或密码不能为空");
            return;
        }
        if (_accounts.ContainsKey(user))
        {
            Debug.Log("注册失败：用户名已存在");
            return;
        }
        _accounts.Add(user, pwd);
        Debug.Log($"注册成功：{user}");
    }

    private void OnLoginClicked()
    {
        string user = _inputAccount.text.Trim();
        string pwd = _inputPassword.text.Trim();
        if (!_accounts.TryGetValue(user, out string saved) || saved != pwd)
        {
            Debug.Log("登录失败：账号或密码错误");
            return;
        }
        Debug.Log($"登录成功：{user}");
        SceneManager.LoadScene(SceneNames.Loading);
    }
}
```

#### 逐行讲解表（摘要）

| 行/块 | 作用 |
|-------|------|
| `_accounts` | 试题6 要求 Dictionary；键值存用户名密码 |
| `Trim()` | 去首尾空格，防误输空格导致登录失败 |
| `TryGetValue` | 比 `ContainsKey`+索引少一次查找 |
| `LoadScene(SceneNames.Loading)` | 成功才跳转；失败不跳 |
| `OnDestroy RemoveListener` | 防重复注册监听 |

### 4.2 LoadingController（5 秒假进度）

#### 本段思路

1. **`Start`** 启协程（不用仅 `OnEnable`）。  
2. `elapsed` 累加 `Time.deltaTime`，`t = elapsed/duration`。  
3. 每帧写 `slider.value = t`，文本写百分比。  
4. `t >= 1` → `LoadScene(SceneNames.Main)`。  
5. `OnDisable` → `StopAllCoroutines()`。

#### 带注释 P-L1 范本

```csharp
using System.Collections;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

/// <summary>
/// W4 Stage1 Loading：5 秒假进度条，满后进入 Main。
/// </summary>
public class LoadingController : MonoBehaviour
{
    [SerializeField] private Slider _slider;
    [SerializeField] private Text _tipText;
    [SerializeField] private float _duration = 5f;

    private Coroutine _loadRoutine;

    private void Start()
    {
        _loadRoutine = StartCoroutine(LoadProgressRoutine());
    }

    private void OnDisable()
    {
        if (_loadRoutine != null)
            StopCoroutine(_loadRoutine);
    }

    private IEnumerator LoadProgressRoutine()
    {
        float elapsed = 0f;
        while (elapsed < _duration)
        {
            elapsed += Time.deltaTime;
            float t = Mathf.Clamp01(elapsed / _duration);
            _slider.value = t;
            if (_tipText != null)
                _tipText.text = $"加载中 {Mathf.RoundToInt(t * 100)}%";
            yield return null;
        }
        SceneManager.LoadScene(SceneNames.Main);
    }
}
```
