### 3.1 LoginController

| # | 操作 |
|:-:|------|
| 1 | `Scripts/LoginController.cs` 新建脚本 |
| 2 | 挂到 **LoginCanvas** 根 |
| 3 | Inspector 拖齐 InputField · Button 引用 |
| 4 | Play：登录成功 → 跳 Main；失败 → Console 输出 |

#### 本段思路

1. `Dictionary<string,string>` 存账号密码。  
2. 登录：Trim 后比对 Dictionary；成功 `LoadScene(1)`；失败 `Debug.Log`。  
3. `OnDestroy` 里 `RemoveListener` 防泄漏。

#### 带注释 P-L1 范本

```csharp
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;

/// <summary>
/// Exam01 登录控制器：Dictionary 存账号，成功跳转主界面。
/// </summary>
public class LoginController : MonoBehaviour
{
    [SerializeField] private InputField _inputAccount;
    [SerializeField] private InputField _inputPassword;
    [SerializeField] private Button _btnLogin;

    // 内存账号表：用户名 -> 密码（考试项目够用；商业用服务端）
    private readonly Dictionary<string, string> _accounts = new Dictionary<string, string>();

    private void Start()
    {
        _btnLogin.onClick.AddListener(OnLoginClicked);
        _accounts.Add("admin", "123456");
    }

    private void OnDestroy()
    {
        _btnLogin.onClick.RemoveListener(OnLoginClicked);
    }

    private void OnLoginClicked()
    {
        string user = _inputAccount.text.Trim();
        string pwd = _inputPassword.text.Trim();
        if (!_accounts.TryGetValue(user, out string savedPwd) || savedPwd != pwd)
        {
            Debug.Log("登录失败：账号或密码错误");
            return;
        }
        Debug.Log($"登录成功：{user}");
        SceneManager.LoadScene(1);
    }
}
```

#### 逐行讲解表

| 行/块 | 作用 |
|-------|------|
| `_accounts` | Dictionary 存账号密码，键=用户名，值=密码 |
| `Trim()` | 去首尾空格，防误输空格导致登录失败 |
| `TryGetValue` | 比 `ContainsKey`+索引少一次查找，更高效 |
| `LoadScene(1)` | 跳转到 Main 场景（Build Index=1） |
| `OnDestroy RemoveListener` | 防重复注册监听导致多次触发 |

#### Inspector 拖引用

| SerializeField | 拖谁 |
|----------------|------|
| `_inputAccount` | `LoginCanvas/LoginPanel/AccountInput` |
| `_inputPassword` | `LoginCanvas/LoginPanel/PasswordInput` |
| `_btnLogin` | `LoginCanvas/LoginPanel/LoginButton` |

- **再挖一层**：`TryGetValue` 返回 bool + 输出参数，比先 `ContainsKey` 再索引少一次哈希查找。

```csharp
// B下一档：添加注册功能
private void OnRegisterClicked()
{
    string user = _inputAccount.text.Trim();
    string pwd = _inputPassword.text.Trim();
    if (string.IsNullOrEmpty(user) || string.IsNullOrEmpty(pwd))
    {
        Debug.Log("注册失败：用户名或密码不能为空");
        return;
    }
    if (_accounts.ContainsKey(user))
    {
        Debug.Log("注册失败：用户名已存在");
        return;
    }
    _accounts.Add(user, pwd);
    Debug.Log($"注册成功：{user}");
}
```

### 5.2 BackpackController

#### 本段思路

1. 创建物品预制体（Image + Text）。  
2. `Start()` 循环 Instantiate 20 个物品到 Content。  
3. 每个物品显示编号和随机名称。  
4. 关闭按钮设置 `SetActive(false)`。

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// Exam01 背包控制器：动态创建20个物品，每行4个，支持滚动。
/// </summary>
public class BackpackController : MonoBehaviour
{
    [SerializeField] private GameObject _itemPrefab;
    [SerializeField] private Transform _content;
    [SerializeField] private Button _btnClose;

    // 物品名称池（随机选取）
    private readonly string[] _itemNames = { "药水", "武器", "护甲", "卷轴", "宝石" };

    private void Start()
    {
        _btnClose.onClick.AddListener(OnCloseClicked);
        CreateItems();
    }

    private void OnDestroy()
    {
        _btnClose.onClick.RemoveListener(OnCloseClicked);
    }

    private void CreateItems()
    {
        for (int i = 0; i < 20; i++)
        {
            GameObject item = Instantiate(_itemPrefab, _content);
            Text itemText = item.GetComponentInChildren<Text>();
            if (itemText != null)
            {
                string name = _itemNames[Random.Range(0, _itemNames.Length)];
```

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// Exam01 技能控制器：管理技能列表和技能弹窗。
/// </summary>
public class SkillController : MonoBehaviour
{
    [SerializeField] private Button[] _skillButtons;
    [SerializeField] private GameObject _skillPopup;
    [SerializeField] private Text _infoText;
    [SerializeField] private Button _btnClosePopup;

    // 四个技能数据
    private readonly SkillData[] _skills = new SkillData[4]
    {
        new SkillData { skillName = "斩击", damage = 10, cooldown = 1f, description = "快速挥砍敌人" },
        new SkillData { skillName = "旋风斩", damage = 25, cooldown = 3f, description = "旋转攻击周围敌人" },
        new SkillData { skillName = "突刺", damage = 15, cooldown = 2f, description = "向前冲刺攻击" },
        new SkillData { skillName = "必杀技", damage = 50, cooldown = 8f, description = "释放全部力量" }
    };

    private void Start()
    {
        _btnClosePopup.onClick.AddListener(ClosePopup);
        for (int i = 0; i < _skillButtons.Length && i < _skills.Length; i++)
        {
            int index = i;
            _skillButtons[i].GetComponentInChildren<Text>().text = _skills[i].skillName;
            _skillButtons[i].onClick.AddListener(() => ShowSkillInfo(index));
        }
    }

    private void OnDestroy()
    {
        _btnClosePopup.onClick.RemoveListener(ClosePopup);
        foreach (var btn in _skillButtons)
        {
            btn.onClick.RemoveAllListeners();
        }
    }

    private void ShowSkillInfo(int index)
    {
        SkillData skill = _skills[index];
        _infoText.text = $"技能名称：{skill.skillName}\n" +
                         $"伤害值：{skill.damage}\n" +
                         $"冷却时间：{skill.cooldown}秒\n" +
                         $"描述：{skill.description}";
        _skillPopup.SetActive(true);
    }

    private void ClosePopup()
    {
        _skillPopup.SetActive(false);
    }
}
```

| `ClosePopup` | 隐藏技能弹窗 |

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// Exam01 技能控制器：管理技能列表、弹窗和动画触发。

public class SkillController : MonoBehaviour
{

    {
```
