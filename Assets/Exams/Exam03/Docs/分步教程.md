### 1.3 PlayerController 脚本

#### 本段思路

1. 获取 CharacterController 组件引用
2. 每帧读取 WASD 输入，计算水平移动方向
3. 处理重力：地面时保持贴地，空中时加速下落
4. 处理跳跃：空格 + 在地面 → 设置向上速度
5. 组合最终速度，调用 Move 方法执行移动

#### 带注释 P-L1 范本

```csharp
using UnityEngine;

/// Exam03 角色控制器：WASD移动 + 空格跳跃 + 重力
public class PlayerController : MonoBehaviour
{
    // --- 移动参数 ---
    [Header("移动")]
    [SerializeField] private float _moveSpeed = 5f;
    [SerializeField] private float _jumpForce = 8f;
    [SerializeField] private float _gravity = 20f;

    // --- 组件引用 ---
    private CharacterController _controller;
    private Vector3 _velocity;

    void Awake()
    {
        // 获取本物体上的 CharacterController 组件
        _controller = GetComponent<CharacterController>();
    }

    void Update()
    {
        // 1. 读取水平输入
        float h = Input.GetAxis("Horizontal");
        float v = Input.GetAxis("Vertical");

        // 2. 转换成角色自身坐标系方向
        Vector3 moveDir = transform.right * h + transform.forward * v;

        // 3. 处理重力与跳跃
        if (_controller.isGrounded)
        {
            // 在地面：保持小的向下速度贴地
            if (_velocity.y < 0) _velocity.y = -2f;

            // 跳跃：按空格且在地面
            if (Input.GetButtonDown("Jump"))
            {
                _velocity.y = _jumpForce;
            }
        }
        else
        {
            // 在空中：重力加速下落
            _velocity.y -= _gravity * Time.deltaTime;
        }

        // 4. 组合最终移动向量
        Vector3 finalMove = moveDir.normalized * _moveSpeed + Vector3.up * _velocity.y;

        // 5. 执行移动（速度 × 时间 = 位移）
        _controller.Move(finalMove * Time.deltaTime);
    }
}
```

### 📌 原理加餐（A）

> 考纲：U11 · 域 2

- **是什么**：CharacterController 是 Unity 为人形角色设计的专用移动组件，自带胶囊碰撞体，不受物理力影响，精确可控
- **卷面为何**：角色移动是游戏最基础也最重要的玩法，必考；重力和跳跃是经典易错点
- **再挖一层**：Move 方法接收的是「本帧位移」不是「速度」，所以必须乘 Time.deltaTime；isGrounded 是上一次 Move 的结果，Move 之后读才准确；重力要自己写，因为 CC 不走物理引擎

```csharp
/// 进阶：状态机驱动的移动 · 第 3 步完整实现
public class PlayerController : MonoBehaviour
{
    [SerializeField] private float _moveSpeed = 5f;
    [SerializeField] private float _jumpForce = 8f;
    [SerializeField] private float _gravity = 20f;

    private StateMachine _stateMachine;
    private PlayerContext _context;

    void Awake()
    {

        _stateMachine = new StateMachine();
        _context = new PlayerContext(
            _stateMachine,
            GetComponent<CharacterController>(),
            GetComponent<Animator>(),
            transform

        _context.MoveSpeed = _moveSpeed;
        _context.JumpForce = _jumpForce;
        _context.Gravity = _gravity;

        // 初始状态：待机
        _stateMachine.Initialize(new IdleState(_context));
    }

    void Update()
    {
        // 读输入到上下文
        _context.MoveInput = new Vector2(
            Input.GetAxis("Horizontal"),
            Input.GetAxis("Vertical")
        );

        // 当前状态更新
        _stateMachine.CurrentState.Update();
    }
}
```

### 2.2 Joystick 脚本

#### 本段思路

1. 实现三个 UI 事件接口：IPointerDownHandler（按下）、IDragHandler（拖拽）、IPointerUpHandler（抬起）
2. 按下时记录位置，立刻响应
3. 拖拽中：屏幕坐标 → UI 局部坐标 → 计算偏移比例 → 死区处理 → 归一化 → 更新 Handle 位置
4. 抬起时：方向归零，Handle 回到中心

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.EventSystems;

/// Exam03 虚拟摇杆：固定位置，输出方向向量
public class Joystick : MonoBehaviour, IPointerDownHandler, IDragHandler, IPointerUpHandler
{
    // --- UI 引用 ---
    [Header("UI 引用")]
    [SerializeField] private RectTransform _background;
    [SerializeField] private RectTransform _handle;

    // --- 参数 ---
    [Header("参数")]
    [SerializeField] [Range(0, 1)] private float _deadZone = 0.1f;
    [SerializeField] [Range(0.5f, 2f)] private float _handleRange = 1f;

    // --- 对外输出 ---
    public Vector2 Direction { get; private set; }

    // --- 单例 ---
    public static Joystick Instance { get; private set; }

    void Awake()
    {
        Instance = this;

        // 自动找引用（防忘拖）
        if (_background == null) _background = GetComponent<RectTransform>();
        if (_handle == null && transform.childCount > 0)
            _handle = transform.GetChild(0).GetComponent<RectTransform>();
    }

    /// 按下：立刻响应一次拖拽
    public void OnPointerDown(PointerEventData eventData)
    {
        OnDrag(eventData);
    }

    /// 拖拽中：每帧计算方向
    public void OnDrag(PointerEventData eventData)
    {
        // 1. 屏幕坐标 → 背景局部坐标
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            _background,
            eventData.position,
            eventData.pressEventCamera,
            out Vector2 localPos);

        // 2. 转成比例（-1 ~ 1）
        Vector2 dir = new Vector2(
            localPos.x / (_background.sizeDelta.x / 2),
            localPos.y / (_background.sizeDelta.y / 2));

        // 3. 死区处理
        if (dir.magnitude < _deadZone)
        {
            Direction = Vector2.zero;
        }
        else
        {
            // 4. 超出范围归一化
            if (dir.magnitude > 1f)
                dir = dir.normalized;

            Direction = dir;
        }

        // 5. 更新 Handle 位置
        _handle.anchoredPosition = Direction * (_background.sizeDelta / 2) * _handleRange;
    }

    /// 抬起：方向归零
    public void OnPointerUp(PointerEventData eventData)
    {
        Direction = Vector2.zero;
        _handle.anchoredPosition = Vector2.zero;
    }
}
```

### 3.1 IState 接口与 StateMachine

#### 本段思路

1. 定义 IState 接口：Enter（进入）、Update（每帧）、Exit（退出）
2. 实现 StateMachine 类：持有当前状态，提供 Initialize 和 ChangeState 方法
3. ChangeState 流程：旧状态 Exit → 切换状态 → 新状态 Enter

#### 带注释 P-L1 范本

```csharp
// ============================================
// IState 接口
// ============================================
/// 状态接口：所有状态都必须实现这三个方法
public interface IState
{
    void Enter();
    void Update();
    void Exit();
}

// ============================================
// StateMachine 状态机管理器
// ============================================
/// 状态机管理器：管理当前状态，负责状态切换
public class StateMachine
{
    public IState CurrentState { get; private set; }

    /// 初始化：设置起始状态并调用 Enter
    public void Initialize(IState startState)
    {
        CurrentState = startState;
        CurrentState.Enter();
    }

    /// 切换状态：旧状态 Exit → 新状态 Enter
    public void ChangeState(IState newState)
    {
        CurrentState.Exit();
        CurrentState = newState;
        CurrentState.Enter();
    }
}
```

### 3.3 Idle / Move / Jump 状态

#### IdleState 待机状态

```csharp
/// 待机状态：无输入时的状态
public class IdleState : IState
{
    private readonly PlayerContext _context;

    public IdleState(PlayerContext context) => _context = context;

    public void Enter()
    {
        // 进入待机：播待机动画
        _context.Animator?.SetBool("IsMoving", false);
    }

    public void Update()
    {
        // 切换条件 1：有输入 → 切到 Move
        if (_context.MoveInput.magnitude > 0.1f)
        {
            _context.StateMachine.ChangeState(new MoveState(_context));
            return;
        }

        // 切换条件 2：按跳跃 → 切到 Jump
        if (Input.GetButtonDown("Jump") && _context.Controller.isGrounded)
        {
            _context.Velocity.y = _context.JumpForce;
            _context.StateMachine.ChangeState(new JumpState(_context));
            return;
        }

        // 待机时：重力下落
        ApplyGravity();
        Move();
    }

    public void Exit() { }

    void ApplyGravity()
    {
        if (_context.Controller.isGrounded && _context.Velocity.y < 0)
            _context.Velocity.y = -2f;
        else
            _context.Velocity.y -= _context.Gravity * Time.deltaTime;
    }

    void Move()
    {
        Vector3 dir = _context.Transform.right * _context.MoveInput.x
                    + _context.Transform.forward * _context.MoveInput.y;
        Vector3 final = dir.normalized * _context.MoveSpeed + Vector3.up * _context.Velocity.y;
        _context.Controller.Move(final * Time.deltaTime);
    }
}
```

#### 本段思路

2. 用 NavMeshAgent.SetDestination 移动到当前路径点
3. 到达后切换到下一个路径点

#### 带注释 P-L1 范本

```csharp
using UnityEngine;
using UnityEngine.AI;

/// Exam03 怪物AI：巡逻 + 追击
public class EnemyAI : MonoBehaviour
{
    // --- 参数 ---
    [Header("移动")]
    [SerializeField] private float _moveSpeed = 3f;
    [SerializeField] private float _stoppingDistance = 2f;

    [Header("感知")]
    [SerializeField] private float _detectionRange = 10f;
    [SerializeField] private float _attackRange = 2f;

    [Header("巡逻路径")]
    [SerializeField] private Transform[] _patrolPoints;

    // --- 组件 ---
    private NavMeshAgent _agent;
    private Transform _player;
    private int _currentPatrolIndex;

    void Awake()
    {
        _agent = GetComponent<NavMeshAgent>();
        _player = GameObject.FindGameObjectWithTag("Player").transform;

        // 设置 Agent 参数
        _agent.speed = _moveSpeed;
        _agent.stoppingDistance = _stoppingDistance;
    }

    void Start()
    {
        // 开始巡逻
        GoToNextPatrolPoint();
    }

    void Update()
    {
        // 检测玩家距离
        float distanceToPlayer = Vector3.Distance(transform.position, _player.position);

        if (distanceToPlayer <= _detectionRange)
        {
            // 发现玩家：追击
            ChasePlayer();
        }
        else
        {
            // 未发现：继续巡逻
            Patrol();
        }
    }

    /// 巡逻：依次前往路径点
    void Patrol()
    {
        // 到达当前路径点
        if (_agent.remainingDistance < _agent.stoppingDistance)
        {
            GoToNextPatrolPoint();
        }
    }

    /// 前往下一个巡逻点
    void GoToNextPatrolPoint()
    {
        if (_patrolPoints.Length == 0) return;

        _currentPatrolIndex = (_currentPatrolIndex + 1) % _patrolPoints.Length;
        _agent.SetDestination(_patrolPoints[_currentPatrolIndex].position);
    }

    /// 追击玩家
    void ChasePlayer()
    {
        _agent.SetDestination(_player.position);

        // 到达攻击范围
        if (_agent.remainingDistance <= _attackRange)
        {
            AttackPlayer();
        }
    }
```

```csharp
/// 进阶：三层感知系统
void Update()
{
    float distance = Vector3.Distance(transform.position, _player.position);

    // 第一层：距离检测（最快）
    if (distance > _detectionRange) return;

    // 第二层：角度检测（中等）
    Vector3 toPlayer = _player.position - transform.position;
    float angle = Vector3.Angle(transform.forward, toPlayer);
    if (angle > _detectionAngle) return;

    // 第三层：射线检测（最精确，检测是否有障碍物）
    RaycastHit hit;
    if (Physics.Raycast(transform.position, toPlayer.normalized, out hit, _detectionRange))
    {
        if (hit.collider.CompareTag("Player"))
        {
            // 真正看到玩家了
            ChasePlayer();
            return;
        }
    }

    // 没看到，继续巡逻
    Patrol();
}
```

### 5.1 MenuItem 批量重命名

#### 本段思路

1. 使用 `[MenuItem]` 属性创建编辑器菜单
2. 获取选中的多个物体
3. 按序号批量重命名

#### 带注释 P-L1 范本

```csharp
using UnityEditor;
using UnityEngine;

/// Exam03 编辑器工具：批量重命名
public class BatchRenameTool
{
    /// 菜单路径：Tools → Exam03 → 批量重命名
    [MenuItem("Tools/Exam03/批量重命名")]
    static void BatchRename()
    {
        // 获取选中的物体
        Object[] selectedObjects = Selection.objects;

        if (selectedObjects.Length == 0)
        {
            EditorUtility.DisplayDialog("提示", "请先选中要重命名的物体", "确定");
            return;
        }

        // 弹出输入对话框，输入前缀
        string prefix = EditorUtility.DisplayDialogComplex(
            "批量重命名",
            "请输入前缀",
            "确定",
            "取消",
            "") == 0 ? EditorUtility.GetString("BatchRenamePrefix", "Object_") : "";

        if (string.IsNullOrEmpty(prefix)) return;

        // 开始重命名
        Undo.RecordObjects(selectedObjects, "批量重命名");

        for (int i = 0; i < selectedObjects.Length; i++)
        {
            selectedObjects[i].name = $"{prefix}{i + 1}";
        }

        EditorUtility.DisplayDialog("完成", $"成功重命名 {selectedObjects.Length} 个物体", "确定");
    }
}
```

```csharp
[CustomEditor(typeof(PlayerController))]
```
