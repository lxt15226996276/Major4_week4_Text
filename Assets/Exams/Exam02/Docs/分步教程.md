### 3.1 PlayerAnim 脚本

#### 本段思路

1. Awake 中获取 Animator 组件，预计算参数哈希（性能优化）
2. Update 中读取输入，计算速度，设置 Float/Bool 参数
3. 检测跳跃输入，用 ResetTrigger + SetTrigger 触发跳跃
4. 使用 StringToHash 避免字符串比较开销

#### 带注释 P-L1 范本

```csharp
/// Exam02 玩家动画控制器：驱动 Animator 参数，控制走跑跳切换
/// [继承 §七] 基础参数设置
/// [递进 vs Exam01] 新增 StringToHash 预计算 + ResetTrigger 防失效
using UnityEngine;

public class PlayerAnim : MonoBehaviour
{
    // --- Animator 组件 ---
    Animator _anim;
    
    // --- 参数哈希（预计算，减少字符串比较开销） ---
    static readonly int SpeedHash = Animator.StringToHash("Speed");
    static readonly int IsGroundedHash = Animator.StringToHash("IsGrounded");
    static readonly int JumpHash = Animator.StringToHash("Jump");
    static readonly int WeaponTypeHash = Animator.StringToHash("WeaponType");

    // --- 运动参数 ---
    [SerializeField] float moveSpeed = 5f;
    [SerializeField] float jumpForce = 8f;
    bool _isGrounded = true;
    Vector3 _velocity;

    /// Awake 初始化：获取 Animator 组件
    void Awake()
    {
        _anim = GetComponent<Animator>();
    }

    /// Update 每帧：读取输入，更新动画参数
    void Update()
    {
        // 同步着地状态到 Animator
        _anim.SetBool(IsGroundedHash, _isGrounded);

        // 读取输入，计算速度
        float h = Input.GetAxisRaw("Horizontal");
        float v = Input.GetAxisRaw("Vertical");
        Vector2 input = new Vector2(h, v);
        float speed = Mathf.Clamp01(input.magnitude);
        
        // 设置速度参数（用哈希）
        _anim.SetFloat(SpeedHash, speed);

        // 移动逻辑
        if (speed > 0.01f)
        {
            Vector3 dir = new Vector3(h, 0, v).normalized;
            transform.position += dir * moveSpeed * Time.deltaTime;
        }

        // 跳跃检测
        if (Input.GetKeyDown(KeyCode.Space) && _isGrounded)
        {
            // 先复位 Trigger，防止连续按失效
            _anim.ResetTrigger(JumpHash);
            _anim.SetTrigger(JumpHash);
            _isGrounded = false;
            _velocity.y = Mathf.Sqrt(jumpForce * -2f * Physics.gravity.y);
        }
    }

    /// FixedUpdate 物理帧：处理垂直运动和落地检测
    void FixedUpdate()
    {
        if (!_isGrounded)
        {
            _velocity.y += Physics.gravity.y * Time.fixedDeltaTime;
            transform.position += _velocity * Time.fixedDeltaTime;

            // 落地检测
            if (transform.position.y <= 0f)
            {
                transform.position = new Vector3(transform.position.x, 0f, transform.position.z);
                _velocity.y = 0f;
                _isGrounded = true;
            }
        }
    }
}
```

### 4.2 StateMachine 类

#### 本段思路

状态机核心类，负责管理当前状态和切换逻辑。ChangeState 方法：先 Exit 旧状态，再 Enter 新状态。

#### 带注释 P-L1 范本

```csharp
/// Exam02 状态机管理器：负责状态切换和驱动
public class StateMachine
{
    IState _currentState;

    /// 切换状态：先退出旧状态，再进入新状态
    /// [递进 vs Exam01] 状态单例化，不每次 new
    public void ChangeState(IState newState)
    {
        _currentState?.Exit();
        _currentState = newState;
        _currentState.Enter();
    }

    /// 每帧驱动当前状态的 Update
    public void Update()
    {
        _currentState?.Update();
    }
}
```

#### 逐行讲解表

| 行/块 | 代码 | 含义 | 为何需要 |
|-------|------|------|----------|
| `_currentState?.Exit()` | 空条件运算符调用 Exit | 旧状态退出，空则跳过 | 第一次切换时无旧状态 |
| `_currentState = newState` | 切换到新状态 | 更新当前状态引用 | 核心逻辑 |
| `_currentState.Enter()` | 调用新状态的 Enter | 初始化新状态 | 完成切换 |
| Update | 驱动当前状态 | 每帧调用 | 状态机核心循环 |

        _currentState.Enter(_context);
    }

    public void Update() => _currentState?.Update(_context);
}
